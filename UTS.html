#import library
import re
import string

#text cleaning
def text_cleaning(text):
  #hapus url
  text = re.compile('<.*?>|&([a-z0-9]+|#[0-9]{1,6}|#x[0-9a-f]{1,6});').sub('', str(text))
 
  #ubah huruf kecil
  text = text.lower()

  #hapus karakter awal dan akhir
  text = re.strip()

  #hapus tanda hubung(punctuation)
  text = re.compile('<.*?>').sub("",text)

  #hapus karakter spesial
  text =re.compile("[%s]" % re.escape(string.punctuation))
  
  #hapus spasi ganda
  text = re.sub("\s+","",text)

  #hapus angka
  text = re.sub(r"\[[0-9]*\]","",text)
  text = re.sub(r"[^\w\s]","",str(text).lower().strip())
  text = re.sub(r"\d","",text)
  text = re.sub(r"\s","",text)

  #ubah text nan menjadi whitespace
  text = re.sub("nan","",text)

  return text

import re

def text_cleaning(text):
  #hapus karakter yang tidak diperlukan
  text = re.sub(r'[^\w\s]','',text)
  
  #hapus angka
  text = re.sub(r'\d+','',text)
  
  #ubah ke huruf kecil
  text = text.lower()
  
  #hapus karakter awal dan akhir
  text = text.strip()
  
  #hapus tanda hubung(punctuation)
  text = re.sub(r'[\-]',' ',text)
  
  #hapus multiple spasi
  text = re.sub(r'\s+',' ',text)
  
  return text
